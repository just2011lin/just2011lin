# 测试手写响应式原理的代码

(function () {
    const testWatch = {
        testProp() {
            const hero = {name: '孙悟空'};
            observe(hero);
            new Watch(hero, 'name', function (newValue, oldValue) {
                doubleAssert(newValue, '猪八戒', oldValue, '孙悟空', '验证一层的属性响应了变化');
            })
            hero.name = '猪八戒';
        },
        testDoubleProp() {
            const hero = {
                master: {
                    name: '唐僧',
                    age: 18
                }
            }
            observe(hero);
            new Watch(hero, 'master.age', function (newValue, oldvalue) {
                doubleAssert(oldvalue, 18, newValue, 24, '验证二层的属性响应了变化');
            });
            hero.master.age = 24;
        },
        testTripleProp() {
            const hero = {
                master: {
                    friend: {
                        age: 18
                    }
                }
            };
            observe(hero);
            new Watch(hero, 'master.friend.age', function (newValue, oldValue) {
                doubleAssert(newValue, 24, oldValue, 18, '验证三层的属性响应了变化');
            })
            hero.master.friend.age = 24;
        },
        testNewObject() {
            const hero = {
                master: null
            }
            observe(hero);
            new Watch(hero, 'master.age', function (newValue, oldValue) {
                doubleAssert(newValue, 18, oldValue, undefined, '验证后设置的对象中的属性的响应');
            });
            hero.master = {age: 18};
        },
        testPushArray() {
            const hero = {
                friends: ['唐三藏']
            };
            observe(hero);
            new Watch(hero, 'friends', function (newValue, oldValue) {
                doubleAssert(newValue[1], '猪八戒', oldValue[0], '唐三藏', '验证数组的push方法');
            });
            hero.friends.push('猪八戒');
        },
        testPushedObject() {
            const hero = {
                friends: []
            };
            observe(hero);
            new Watch(hero, 'friends.0.age', function (newValue, oldValue) {
                doubleAssert(newValue, 18, oldValue, undefined, '验证用push方法添加的对象的属性变化');
            })
            hero.friends.push({age: 18})
        },
        testNestedArray() {
            const hero = {
                friends: [['张三丰']]
            }
            observe(hero);
            new Watch(hero, 'friends', function (newValue, oldValue) {
                doubleAssert(newValue[0][1], '孙悟空', oldValue[0][0], '张三丰', '验证二位数组中内数组的变化触发了外数组的监听');
            });
            hero.friends[0].push('孙悟空');
        },
        testArrayObject() {
            const hero = {
                friends: [{age: 18}]
            };
            observe(hero);
            new Watch(hero, 'friends.0.age', function (newValue, oldValue) {
                doubleAssert(newValue, 24, oldValue, 18, '验证数组中对象属性的监听');
            })
            hero.friends[0].age = 24;
        }
    }

    let length: number;
    let index = 0;

    function doubleAssert(a, b, c, d, message) {
        index++;
        if (a === b && c === d) {
            console.log('通过：', message);
        } else {
            console.error('不通过：', message);
        }
    }

    const values = Object.values(testWatch);
    length = values.length;
    for (let func of values) {
        func();
    }
    if (index !== length) {
        console.error('测试了', length, '个方法，有', length - index, '个方法未响应');
    } else {
        console.log('测试了', length, '个方法，没有方法未响应');
    }
})();
