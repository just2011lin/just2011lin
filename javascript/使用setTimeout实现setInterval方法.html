      # 使用setTimeout实现setInterval方法

      # 完整代码

      /**
* setMyInterval启动后用于清除定时器的数据类型
*/
type MyIntervalId = { timeout: number };

/**
* 使用类来记录setTimtout的返回值
*/
class MyInterval {
   // 记录setTimout的返回值，用于终止循环
   private myIntervalId: MyIntervalId = {timeout: undefined};

   setInterval(callback: (...args: any[]) => any, delay: number, ...args: any[]): MyIntervalId {
       // @ts-ignore
       this.myIntervalId.timeout = setTimeout(() => {
           // 移除timeout
           callback(...args);
           this.setInterval(callback, delay, ...args);
       }, delay);
       return this.myIntervalId;
   }
}

/**
* 设置我的循环定时器函数
* @param callback 执行的函数
* @param delay 延迟执行的时间
* @param args 传入执行函数的参数
*/
export function setMyInterval(callback: (...args: any[]) => any, delay: number, ...args: any[]): MyIntervalId {
   return new MyInterval().setInterval(callback, delay, ...args);
}

/**
* 终止我的循环定时器函数
* @param myIntervalId
*/
export function clearMyInterval(myIntervalId: MyIntervalId): void {
   clearTimeout(myIntervalId.timeout);
}

      # 附言

      1. 使用MyIntervalId而不是直接返回MyInterval类

      本质上可以设置MyInterval类myIntervalId为数值类型，并将setTimeout方法的返回值赋值给myInterval， 然后在实例方法setInterval中返回类本身。

      这样做的话，用户将可以直接操作MyInterval实例，如果用户误用了实例方法setInterval，将可能导致无法预估的问题， 所以采用了返回一个其他类型的值的方式

      这样的开发原则，我应该取一个名称，就叫：代码权限限制原则
