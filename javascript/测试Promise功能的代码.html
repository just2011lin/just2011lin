# 测试Promise的代码

使用方法：引入此脚本，调用testPromise方法并传入自定义的Promise类，然后在控制台查看是否通过

export function testPromise(CustomPromise: any): void {
    Object.values(testFuncs).forEach(func => func(CustomPromise));
}

function assertFunc(a: any, b: any, funcName: string): void {
    if (a === b) {
        console.log(funcName + ': 测试通过');
    } else {
        console.error(funcName + ': 测试未通过');
    }
}

const testFuncs = {
    // 测试：在resolve后可以获取值
    resolveTest(CustomPromise: any): void {
        new CustomPromise((resolve: any, _: any) => {
            resolve(1);
        }).then((value: any) => {
            assertFunc(value, 1, 'resolveTest')
        }, undefined);
    },
    // 测试：在定时器中resolve后可以取得值
    timeoutResolveTest(CustomPromise: any): void {
        new CustomPromise((resolve: any, _: any) => {
            setTimeout(() => {
                resolve(1);
            }, 1000);
        }).then((value: any) => {
            assertFunc(value, 1, 'timeoutResolveTest');
        }, undefined)
    },
    // 测试：使用reject并可以取得reason
    rejectTest(CustomPromise: any): void {
        new CustomPromise((_: any, reject: any) => {
            reject('错了');
        }).then(undefined, (reason: any) => {
            assertFunc(reason, '错了', 'rejectTest');
        })
    },
    // 测试：timeout后reject并可以取得reason
    timeoutRejectTest(CustomPromise: any): void {
        new CustomPromise((_: any, reject: any) => {
            setTimeout(() => {
                reject('错了');
            }, 1000);
        }).then(undefined, (reason: any) => {
            assertFunc(reason, '错了', 'timeoutRejectTest');
        })
    },
    // 测试：经历两次then仍然能够取到值
    twiceResolveTest(CustomPromise: any): void {
        new CustomPromise((resolve: any, _: any) => {
            setTimeout(() => {
                resolve(1);
            }, 2000)
        }).then((value: any) => value + 1, undefined).then((value: any) => {
            assertFunc(value, 2, 'twiceResolveTest');
        }, undefined);
    },
    // 测试：onRejected返回的值在下一个then中取到
    getOnRejectedTest(CustomPromise: any): void {
        new CustomPromise((_: any, reject: any) => {
            setTimeout(() => {
                reject('错了');
            }, 2000);
        }).then(undefined, (reason: any) => {
            return reason + '才怪';
        }).then((value: any) => {
            assertFunc(value, '错了才怪', 'getOnRejectedTest');
        }, undefined);
    },
    // 测试：在onResolved中返回一个MyPromise实例并获取值
    onResolvedMyPromiseTest(CustomPromise: any): void {
        new CustomPromise((resolve: any, _: any) => {
            setTimeout(() => {
                resolve(1);
            }, 1000);
        }).then((value: any) => {
            return new CustomPromise((resolve: any, _: any) => {
                setTimeout(() => {
                    resolve(value + 1);
                }, 2000);
            })
        }, undefined).then((value: any) => {
            assertFunc(value, 2, 'onResolvedMyPromiseTest');
        }, undefined);
    },
    // 测试：在onRejected中返回一个MyPromise实例并取值
    onRejectedPromiseTest(CustomPromise: any): void {
        new CustomPromise((_: any, reject: any) => {
            setTimeout(() => {
                reject('错了');
            }, 1000);
        }).then(undefined, (reason: any) => {
            return new CustomPromise((_: any, reject: any) => {
                setTimeout(() => {
                    reject(reason + '就是错了');
                }, 2000);
            })
        }).then(undefined, (reason: any) => {
            assertFunc(reason, '错了就是错了', 'onRejectedPromiseTest');
        })
    },
    // 测试：多次订阅同一个Promise并都可以获取resolve的值
    doubleResolveThenTest(CustomPromise: any): void {
        const myPromise = new CustomPromise((resolve: any, _: any) => {
            setTimeout(() => {
                resolve(1);
            }, 1000);
        });
        myPromise.then((value: any) => {
            myPromise.then((value2: any) => {
                assertFunc(value2, value, 'doubleResolveThenTest');
            }, undefined);
        }, undefined);
    },
    // 测试：多次订阅同一个Promise并都可以获取reject的理由
    doubleRejectThenTest(CustomPromise: any): void {
        const myPromise = new CustomPromise((_: any, reject: any) => {
            setTimeout(() => {
                reject('错了');
            }, 1000);
        });
        myPromise.then(undefined, (reason: any) => {
            myPromise.then(undefined, (reason2: any) => {
                assertFunc(reason, reason2, 'doubleRejectThenTest');
            })
        })
    },
    // 测试：若在executor函数内报错，则MyPromise转为rejected状态
    executorThrowErrorTest(CustomPromise: any): void {
        new CustomPromise((_: any, __: any) => {
            throw '错了';
        }).then(undefined, (reason: any) => {
            assertFunc(reason, '错了', 'executorThrowErrorTest');
        });
    },
}
