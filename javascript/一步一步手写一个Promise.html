  1. 最终代码
type OneParamFunc = (value: any) => any;
type NilParamFunc = () => any;
type MyPromiseExecutorFunc = (resolve: OneParamFunc, reject: OneParamFunc) => void;

class MyPromise {
    // 三个状态
    private status: 'pending' | 'resolved' | 'rejected' = 'pending';
    // resolved的值
    private value: any;
    // rejected的理由
    private reason: any;
    // resolve之后的回调
    private onResolvedCalls: NilParamFunc[] = [];
    // reject之后的回调
    private onRejectedCalls: NilParamFunc[] = [];
    // 不处理reason异常就提示的timer
    private handleReasonTimer: any = undefined;

    constructor(executor: MyPromiseExecutorFunc) {
        try {
            executor(this.resolve, this.reject);
        } catch (error) {
            this.reject(error);
        }
    }

    private resolve: OneParamFunc = (value: any): void => {
        if (this.status === 'pending') {
            this.status = 'resolved';
            this.value = value;
            this.onResolve();
        }
    }

    private reject: OneParamFunc = (reason: any): void => {
        if (this.status === 'pending') {
            this.status = 'rejected';
            this.reason = reason;
            this.handleReasonTimer = setTimeout(() => {
                console.error('未处理异常：', this.reason);
            });
            this.onReject();
        }
    }

    private onResolve(): void {
        this.onResolvedCalls.forEach(call => {
            if (typeof call === 'function') {
                window.queueMicrotask(call);
            }
        });
    }

    private onReject(): void {
        this.onRejectedCalls.forEach(call => {
            clearTimeout(this.handleReasonTimer);
            if (typeof call === 'function') {
                window.queueMicrotask(call);
            }
        });
    }

    private pushOnResolveRejectCall(onResolveCall: NilParamFunc, onRejectCall: NilParamFunc): void {
        switch (this.status) {
            case "pending":
                this.onResolvedCalls.push(onResolveCall);
                this.onRejectedCalls.push(onRejectCall);
                break;
            case "resolved":
                window.queueMicrotask(onResolveCall);
                break;
            case "rejected":
                window.queueMicrotask(onRejectCall);
                break;
            default:
                break;
        }
    }

    then(onResolved: OneParamFunc, onRejected: OneParamFunc): MyPromise {
        return new MyPromise((resolve, reject) => {
            this.pushOnResolveRejectCall(() => {
                if (typeof onResolved === 'function') {
                    const onResolvedValue = onResolved(this.value);
                    if (onResolvedValue instanceof MyPromise) {
                        onResolvedValue.then(resolve, reject);
                    } else {
                        resolve(onResolvedValue);
                    }
                } else {
                    resolve(this.value);
                }
            }, () => {
                if (typeof onRejected === 'function') {
                    clearTimeout(this.handleReasonTimer);
                    const onRejectedValue = onRejected(this.reason);
                    if (onRejectedValue instanceof MyPromise) {
                        onRejectedValue.then(resolve, reject);
                    } else {
                        resolve(onRejectedValue);
                    }
                } else {
                    reject(this.reason);
                }
            });
        });
    }
}
