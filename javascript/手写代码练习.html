# 手写代码练习

  • Function.prototype.bind
const myBindSymbol = Symbol('myBindSymbol');

function myBind(func: Function, thisArg: any, ...args: any[]): Function {
    return function (...resArgs: any[]): any {
        let res: any;
        if (thisArg) {
            // 绑定this与参数的功能
            thisArg[myBindSymbol] = func;
            res = thisArg[myBindSymbol](...args, ...resArgs);
            delete thisArg[myBindSymbol];
        } else {
            // 只绑定参数的功能
            res = func(...args, ...resArgs);
        }
        return res;
    }
}

const myCallSymbol = Symbol('myCallSymbol');

function myCall(func: Function, thisArg: any, ...args: any[]): any {
    if (thisArg) {
        thisArg[myCallSymbol] = func;
        const
            res = thisArg[myCallSymbol](...args);
        delete thisArg[myCallSymbol];
        return res;
    } else {
        return func(...args);
    }
}
  • Function.prototype.apply
const myApplySymbol = Symbol('myApplySymbol');

function myApply(func: Function, thisArg: any, args?: any[]): any {
    let newArgs = args || [];
    if (thisArg) {
        thisArg[myApplySymbol] = func;
        const res = thisArg[myApplySymbol](...newArgs);
        delete thisArg[myApplySymbol];
        return res;
    } else {
        return func(...newArgs);
    }
}

function unique(arr: any[]): any[] {
    const uniqueArr = [];
    for (let value of arr) {
        if (uniqueArr.indexOf(value) === -1) {
            uniqueArr.push(value);
        }
    }
    return uniqueArr;
}
