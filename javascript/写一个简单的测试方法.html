    # 一个简单的测试方法

    type AssertFunc = (a: any, b: any) => void;

// 测试内容类型
interface TestFuncs {
    [key: string]: (assert: AssertFunc) => void;
}

export function customTest(testFuncs: TestFuncs) {
    // 取出对象中的方法
    const testFuncsEntries = Object.entries(testFuncs).filter(([_, testFunc]) => typeof testFunc === 'function');
    // 记录已测试完成的方法数
    let n = 0;
    // 记录未通过测试的方法
    let wrongKeys: string[] = [];
    // 记录待测试方法数目
    let length = testFuncsEntries.length;
    for (let [key, testFunc] of Object.entries(testFuncs)) {
        testFunc((a: any, b: any) => {
            n++;
            if (a !== b) {
                wrongKeys.push(key);
            }
            if (n === length) {
                console.log(`共测试了${length}个方法`);
                if (wrongKeys.length === 0) {
                    console.log('全部通过测试');
                } else {
                    console.log(`未通过测试的方法：${wrongKeys.join(',')}`);
                }
            }
        });
    }
}
