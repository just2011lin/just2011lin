        # 实现一个简单的长列表

        # LongList Component代码

        import React, {ReactElement} from "react";
import './long-list.less';

interface LongListProps {
    list: any[];
}

interface LongListState {
    startIndex: number;
    endIndex: number;
}

export class LongList extends React.Component<LongListProps, LongListState> {

    heightStyle = {
        height: (this.props.list.length * 41 - 400) + 'px'
    }

    constructor(props: LongListProps) {
        super(props);
        this.state = {
            startIndex: 0,
            endIndex: Math.min(props.list.length, 12),
        };
    }

    getRnderList(): any[] {
        const {startIndex, endIndex} = this.state;
        return this.props.list.slice(startIndex, endIndex);
    }

    mapItem(item: any): ReactElement {
        return (
            <li key={String(item)}>{item}</li>
        )
    }

    handleScroll = (event: React.UIEvent<HTMLDivElement>) => {
        const scrollTop = (event.currentTarget as HTMLDivElement).scrollTop;
        const startIndex = Math.floor(scrollTop / 41);
        const endIndex = startIndex + 12;
        this.setState({
            startIndex, endIndex
        })
    }

    render() {
        const renderList = this.getRnderList();
        return (
            <div className="LongList" onScroll={event => this.handleScroll(event)}>
                {/* 具体内容渲染在此处 */}
                <ul>
                    {renderList.map(this.mapItem)}
                </ul>
                {/* 用这个元素来撑开高度 */}
                <div className="LongList-height" style={this.heightStyle}></div>
            </div>
        )
    }
}

        # 组件样式

        .LongList {
    height: 400px;
    width: 200px;
    overflow-y: auto;

    ul {
        height: 400px;
        position: sticky;
        top: 0;
    }

    li {
        list-style: none;
        line-height: 40px;
        border-bottom: 1px solid #666666;
    }
}
